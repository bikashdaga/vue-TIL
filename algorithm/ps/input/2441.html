# BOJ 2441 별찍기 - 4

# Solution

const readline = require('readline');
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

let count = 0;
let arr = [];

rl.on('line', (line) => {
    count = +line;
    rl.close();
}).on('close', () => {
    for (let i = 0; i < count; i++) {
        arr.push('');

        for (let j = 0; j < i; j++) {
            arr[i] += ' ';
        }

        for (j = i; j < count; j++) {
            arr[i] += '*';
        }
    }
    console.log(arr.join('\n'));
    process.exit();
});
